package com.vocal.triunity.activities;

import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.ImageButton;

import com.vocal.triunity.R;

public class AboutTriunityBuyer extends AppCompatActivity {

    ImageButton button;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_about_triunity_buyer);
        button=findViewById(R.id.back_btn5);


        button.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                startActivity(new Intent(AboutTriunityBuyer.this,MainBuyerActivity.class));
                finish();
            }
        });
    }

}
