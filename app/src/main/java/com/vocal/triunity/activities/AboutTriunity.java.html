package com.vocal.triunity.activities;

import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.ImageButton;

import com.vocal.triunity.R;

public class AboutTriunity extends AppCompatActivity {


    ImageButton backBtn;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_about_triunity);
        backBtn=findViewById(R.id.back_btn4);


            backBtn.setOnClickListener(new View.OnClickListener() {
                @Override
                public void onClick(View v) {
                    startActivity(new Intent(AboutTriunity.this,MainSellerActivity.class));
                    finish();
                }
            });

    }
}
